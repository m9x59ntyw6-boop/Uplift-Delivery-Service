⚙️ 1. Backend (Node.js + Socket.IO)

Create a file: server.js

const express = require("express");
const http = require("http");
const { Server } = require("socket.io");
const cors = require("cors");

const app = express();
app.use(cors());

const server = http.createServer(app);

const io = new Server(server, {
  cors: {
    origin: "*",
  },
});

// Store users
let users = {};

io.on("connection", (socket) => {
  console.log("User connected:", socket.id);

  // Register user (customer or delivery)
  socket.on("register", ({ userId, role }) => {
    users[userId] = socket.id;
    console.log(`${role} registered: ${userId}`);
  });

  // Send message
  socket.on("sendMessage", ({ senderId, receiverId, message }) => {
    const receiverSocket = users[receiverId];

    if (receiverSocket) {
      io.to(receiverSocket).emit("receiveMessage", {
        senderId,
        message,
        time: new Date(),
      });
    }
  });

  socket.on("disconnect", () => {
    console.log("User disconnected:", socket.id);
  });
});

server.listen(3000, () => {
  console.log("Server running on port 3000");
});

Chat Screen (ChatScreen.js)

import React, { useEffect, useState } from "react";
import {
  View,
  Text,
  TextInput,
  TouchableOpacity,
  FlatList,
  StyleSheet,
} from "react-native";
import io from "socket.io-client";

const socket = io("http://YOUR_SERVER_IP:3000");

export default function ChatScreen() {
  const [message, setMessage] = useState("");
  const [messages, setMessages] = useState([]);

  const senderId = "customer_1";
  const receiverId = "delivery_1";

  useEffect(() => {
    socket.emit("register", { userId: senderId, role: "customer" });

    socket.on("receiveMessage", (data) => {
      setMessages((prev) => [...prev, data]);
    });

    return () => socket.disconnect();
  }, []);

  const sendMessage = () => {
    if (message.trim() === "") return;

    const msgData = {
      senderId,
      receiverId,
      message,
    };

    socket.emit("sendMessage", msgData);

    setMessages((prev) => [...prev, msgData]);
    setMessage("");
  };

  const renderItem = ({ item }) => (
    <View
      style={[
        styles.messageBubble,
        item.senderId === senderId
          ? styles.myMessage
          : styles.otherMessage,
      ]}
    >
      <Text style={styles.text}>{item.message}</Text>
    </View>
  );

  return (
    <View style={styles.container}>
      <FlatList
        data={messages}
        renderItem={renderItem}
        keyExtractor={(_, i) => i.toString()}
      />

      <View style={styles.inputContainer}>
        <TextInput
          value={message}
          onChangeText={setMessage}
          placeholder="Type a message..."
          style={styles.input}
        />
        <TouchableOpacity onPress={sendMessage}>
          <Text style={styles.send}>Send</Text>
        </TouchableOpacity>
      </View>
    </View>
  );
}

⚙️ 2. Setup Notification Permissions + Token

Add this to your app (e.g., App.js or inside your chat screen):

import * as Notifications from "expo-notifications";
import * as Device from "expo-device";
import { useEffect } from "react";
import * as Permissions from "expo-permissions";

async function registerForPushNotificationsAsync() {
  if (!Device.isDevice) {
    alert("Must use physical device");
    return;
  }

  const { status: existingStatus } =
    await Notifications.getPermissionsAsync();

  let finalStatus = existingStatus;

  if (existingStatus !== "granted") {
    const { status } =
      await Notifications.requestPermissionsAsync();
    finalStatus = status;
  }

  if (finalStatus !== "granted") {
    alert("Permission not granted!");
    return;
  }

  const token = (await Notifications.getExpoPushTokenAsync()).data;
  console.log("Push Token:", token);

  return token;
}

useEffect(() => {

  socket.on("receiveMessage", (data) => {
    setMessages((prev) => [...prev, data]);

    // 🔔 Show notification when message received
    Notifications.scheduleNotificationAsync({
      content: {
        title: "New Message",
        body: data.message,
      },
      trigger: null,
    });
  });

  return () => socket.disconnect();
}, []);

🌐 4. Send Notifications From Backend (REAL Push)

For real production (when app is closed), your server should send push notifications.

Update your server.js:

const fetch = require("node-fetch");

// Store push tokens
let pushTokens = {};

io.on("connection", (socket) => {
  socket.on("register", ({ userId, role, pushToken }) => {
    users[userId] = socket.id;
    pushTokens[userId] = pushToken;
  });

  socket.on("sendMessage", async ({ senderId, receiverId, message }) => {
    const receiverSocket = users[receiverId];

    if (receiverSocket) {
      io.to(receiverSocket).emit("receiveMessage", {
        senderId,
        message,
      });
    }

    // 🔔 Send push notification
    const token = pushTokens[receiverId];

    if (token) {
      await fetch("https://exp.host/--/api/v2/push/send", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          to: token,
          sound: "default",
          title: "New Message",
          body: message,
        }),
      });
    }
  });
});
